## 14.通过使用 Formik 进行表单控件处理

- 使用 Formik 来实现一个简单的表单处理程序

```tsx
import React, { Component } from 'react';
import { Formik, Form, Field, ErrorMessage } from 'formik';

class HookCom extends Component<any, any> {
  render() {
    return (
      <React.Fragment>
        <Formik
          // 这里定义了控件的初始值
          initialValues={{ email: '', password: '', age: 0 }}
          // 在这里做表单控件值的校验
          validate={(values) => {
            const errors: any = {};
            if (!values.email) errors.email = '邮箱必填';

            if (!values.password) errors.email = '密码必填';
            return errors;
          }}
          // 这里是表单提交后的逻辑
          onSubmit={(values, { setSubmitting }) => {
            setTimeout(() => {
              console.log('提交的values', values);
              // 设置是否处于提交中的状态
              setSubmitting(false);
            }, 2000);
          }}
        >
          {({ isSubmitting }) => (
            <Form>
              <Field type="number" name="age"></Field>
              <Field type="email" name="email"></Field>
              <ErrorMessage name="email" component="div" />
              <Field type="password" name="password" />
              <ErrorMessage name="password" component="div" />
              <button style={{ background: 'red' }} type="submit" disabled={isSubmitting}>
                提交
              </button>
            </Form>
          )}
        </Formik>
      </React.Fragment>
    );
  }
}

export default HookCom;
```
