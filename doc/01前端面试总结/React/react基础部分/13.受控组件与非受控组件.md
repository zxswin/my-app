## 受控组件与非受控组件

- 受控组件

```tsx
/**
 * 什么是受控组件
 * React表单控件的可变状态,通常保存在组件的state中,并且只能通过setState()来更新
 * 受控组件上指定value的prop会阻止用户的更改输入。
 * 如果你指定了 value，但输入仍可编辑，则可能是你意外地将value 设置为 undefined 或 null
 */
class NameForm extends React.Component {
  constructor(props) {
    super(props);
    this.state = { value: '' };

    this.handleChange = this.handleChange.bind(this);
    this.handleSubmit = this.handleSubmit.bind(this);
  }

  handleChange(event) {
    this.setState({ value: event.target.value });
  }

  handleSubmit(event) {
    alert('提交的名字: ' + this.state.value);
    event.preventDefault();
  }

  render() {
    return (
      <form onSubmit={this.handleSubmit}>
        <label>
          名字:
          <input type="text" value={this.state.value} onChange={this.handleChange} />
        </label>
        <input type="submit" value="提交" />
      </form>
    );
  }
}
```

- 非受控组件

```tsx
/**
 * 使用非受控组件来管理表单数据,这时表单数据将交由DOM节点来处理
 * 可以使用ref来从DOM节点中获取表单数据
 * 表单控件可以通过指定一个defaultValue的属性来指定默认值
 * type="checkbox" type="radio" 支持使用defaultChecked属性
 */
class NameForm extends React.Component {
  constructor(props) {
    super(props);
    this.input = React.createRef();
  }

  handleSubmit = (event) => {
    alert('A name was submitted: ' + this.input.current.value);
    event.preventDefault();
  };

  render() {
    return (
      <form onSubmit={this.handleSubmit}>
        <label>
          Name:
          <input type="text" ref={this.input} />
        </label>
        <input type="submit" value="Submit" />
      </form>
    );
  }
}
```
