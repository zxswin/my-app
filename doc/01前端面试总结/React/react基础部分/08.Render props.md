## Render props

```tsx
/**
 * render prop
 * 指React 组件之间使用一个值为函数的prop共享代码(状态数据)的技术
 * 这个传入渲染组件的props可以是其他属性名称,可以不是render
 */

import React, { Component } from 'react';

class Cat extends React.Component<any, any> {
  render() {
    const mouse = this.props.mouse;
    return (
      <div style={{ position: 'absolute', border: '1px solid #f00', width: '100px', height: '100px', left: mouse.x, top: mouse.y }}></div>
    );
  }
}

class Mouse extends Component<any, any> {
  constructor(props) {
    super(props);
    this.state = {
      x: 0,
      y: 0,
    };
  }

  handleOnMouseMove = (event: React.MouseEvent<HTMLDivElement, MouseEvent>) => {
    this.setState({
      x: event.clientX,
      y: event.clientY,
    });
  };

  render() {
    return (
      <React.Fragment>
        <div style={{ height: '100vh' }} onMouseMove={this.handleOnMouseMove}></div>

        {/** 在这里执行Render props 渲染出需要使用到共享状态的组件  */}
        {this.props.render(this.state)}
      </React.Fragment>
    );
  }
}

class Game extends Component<any, any> {
  render() {
    return (
      <React.Fragment>
        <h1>鼠标坐标</h1>
        {/** 通过render的属性传入一个需要使用共享状态渲染的组件逻辑的函数,
         * 这个函数将会在Mouse组件中执行并渲染出对应的组件
         *
         */}
        <Mouse render={(mouse) => <Cat mouse={mouse} />} />
      </React.Fragment>
    );
  }
}

export default Game;
```
